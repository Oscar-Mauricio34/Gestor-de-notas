INICIO

    
    notas = []            # Lista para guardar las notas
    sumaNotas = 0         # Acumulador de notas
    contador = 0          # Contador de notas
    
    # Entrada de datos
    ESCRIBIR "Ingrese una nota (o -1 para terminar): "
    LEER nota
    
    MIENTRAS nota <> -1 HACER
        SI nota >= 0 Y nota <= 100 ENTONCES
            notas[contador] = nota      # Guardar la nota
            sumaNotas ← sumaNotas + nota
            contador ← contador + 1
        SINO
            ESCRIBIR "Nota inválida, ingrese un valor entre 0 y 100."
        FIN_SI
        
        ESCRIBIR "Ingrese otra nota (o -1 para terminar): "
        LEER nota
    FIN_MIENTRAS
    
    # Calcular y mostrar resultados
    SI contador > 0 ENTONCES
        promedio = sumaNotas / contador
        ESCRIBIR "Se ingresaron ", contador, " notas."
        ESCRIBIR "El promedio de notas es: ", promedio
    SINO
        ESCRIBIR "No se ingresaron notas válidas."
    FIN_SI
FIN
